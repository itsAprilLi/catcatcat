function drawSelfPortrait(x, y) {
  push(); translate(x, y);
  noStroke(); fill('#FFD700');
  ellipse(0, 0, 220, 280); // head
  fill('#A0522D');
  beginShape(); // hair
  vertex(-110, -140); bezierVertex(-140, -60, -100, 100, -90, 150);
  bezierVertex(-20, 160, 20, 160, 90, 150);
  bezierVertex(100, 100, 140, -60, 110, -140); endShape(CLOSE);
  drawCoolEye(-50, -30); drawCoolEye(50, -30);
  fill('#FFB347'); ellipse(0, 20, 15, 10); // nose
  fill('#FF69B4'); arc(0, 50, 60, 30, 0, PI); // mouth
  pop();
}
