portrait

function drawSelfPortrait(x, y) {
  push(); translate(x, y); noStroke();
  fill('#FFD700'); ellipse(0, 0, 220, 280);
  fill('#A0522D');
  beginShape();
  vertex(-110, -140);
  bezierVertex(-140, -60, -100, 100, -90, 150);
  bezierVertex(-20, 160, 20, 160, 90, 150);
  bezierVertex(100, 100, 140, -60, 110, -140);
  endShape(CLOSE);
  drawCoolEye(-50, -30);
  drawCoolEye(50, -30);
  fill('#FFB347'); ellipse(0, 20, 15, 10);
  fill('#FF69B4'); arc(0, 50, 60, 30, 0, PI);
  pop();
}

function drawCatHead(x, y) {
  push(); translate(x, y); fill('#4E63FC');
  ellipse(0, 0, 300, 200);
  triangle(-80, -60, -50, -140, -20, -60);
  triangle(80, -60, 50, -140, 20, -60);
  drawEye(-70, -10);
  drawEye(70, -10);
  drawMouth(0, 50); pop();
}
